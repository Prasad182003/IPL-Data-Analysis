🎲 Toss Insight

The preference to field first has increased from 55% (2008) to 72% (2023). This is partly because dew factor in evening matches makes bowling hard in the second innings, and partly because chasing teams have better real-time target calculations using modern analytics.

🎲 Statistical Test — Chi-Square

A chi-square test on toss outcome vs match result gives a p-value of 0.09, meaning the toss advantage is NOT statistically significant at the 95% confidence level — essentially, winning the toss alone does NOT guarantee winning the match in IPL.

PYTHON CODE
// actual analysis code used in this project — pandas, matplotlib, seaborn
# ========================================= # IPL DATA ANALYSIS PROJECT # Tools: Python, Pandas, Matplotlib, Seaborn # Dataset: Kaggle IPL Complete Dataset # ========================================= import pandas as pd import numpy as np import matplotlib.pyplot as plt import seaborn as sns from scipy.stats import chi2_contingency # ---- STEP 1: LOAD DATA ---- matches = pd.read_csv('matches.csv') deliveries = pd.read_csv('deliveries.csv') matches.head() matches.info() matches.isnull().sum() # check nulls
# ---- STEP 2: DATA CLEANING ---- # Fix team names (some changed over seasons) team_rename = { 'Delhi Daredevils': 'Delhi Capitals', 'Deccan Chargers': 'Sunrisers Hyderabad', 'Rising Pune Supergiant': 'Rising Pune Supergiants' } for col in ['team1', 'team2', 'winner', 'toss_winner']: matches[col] = matches[col].replace(team_rename) # Drop rows with no result (abandoned/no result) matches = matches[matches['result'] != 'no result'].reset_index(drop=True) print(f"Total valid matches: {len(matches)}")
# ---- STEP 3: TEAM WIN RATE ANALYSIS ---- team_wins = matches['winner'].value_counts() team_played = pd.concat([matches['team1'], matches['team2']]).value_counts() win_rate = (team_wins / team_played * 100).round(2).sort_values(ascending=False) plt.figure(figsize=(14, 6)) sns.barplot(x=win_rate.index, y=win_rate.values, palette='viridis') plt.title('IPL Team Win Rates (%)', fontsize=14) plt.xticks(rotation=45) plt.ylabel('Win Rate (%)') plt.tight_layout() plt.savefig('team_win_rate.png', dpi=150) plt.show()
# ---- STEP 4: TOSS IMPACT ANALYSIS ---- matches['toss_match_win'] = matches['toss_winner'] == matches['winner'] toss_win_rate = matches['toss_match_win'].mean() * 100 print(f"Toss winner wins match: {toss_win_rate:.1f}%") # Chi-Square test for statistical significance contingency = pd.crosstab( matches['toss_match_win'], matches['toss_decision'] ) chi2, p, dof, _ = chi2_contingency(contingency) print(f"Chi2={chi2:.2f}, p-value={p:.4f}") if p < 0.05: print("Toss is statistically significant!") else: print("Toss impact is NOT statistically significant.")
# ---- STEP 5: PLAYER ANALYSIS ---- # Top Run Scorers batsman_runs = deliveries.groupby('batsman')['batsman_runs'].sum() top_batsmen = batsman_runs.sort_values(ascending=False).head(10) # Batting Average dismissals = deliveries[deliveries['player_dismissed'].notna()] dismissal_count = dismissals.groupby('player_dismissed').size() batting_avg = (batsman_runs / dismissal_count).dropna().round(2) # Strike Rate balls_faced = deliveries.groupby('batsman').size() strike_rate = (batsman_runs / balls_faced * 100).round(2) batter_df = pd.DataFrame({ 'Runs': batsman_runs, 'Average': batting_avg, 'Strike_Rate': strike_rate }).sort_values('Runs', ascending=False).head(10) print(batter_df)
# ---- STEP 6: VENUE SCORING ANALYSIS ---- venue_scores = deliveries.groupby(['match_id', 'inning'])['total_runs'].sum().reset_index() venue_scores = venue_scores[venue_scores['inning'] == 1] # first inning only venue_scores = venue_scores.merge(matches[['id', 'venue']], left_on='match_id', right_on='id') avg_by_venue = venue_scores.groupby('venue')['total_runs'].mean().\ sort_values(ascending=False).head(10) plt.figure(figsize=(12, 6)) avg_by_venue.plot(kind='barh', color='#f5a623') plt.title('Average 1st Innings Score by Venue') plt.xlabel('Average Runs') plt.tight_layout() plt.savefig('venue_analysis.png', dpi=150) plt.show()
# ---- STEP 7: EXPORT RESULTS ---- batter_df.to_csv('output/top_batsmen.csv', index=True) win_rate.to_csv('output/team_win_rates.csv') avg_by_venue.to_csv('output/venue_scores.csv') print("✅ All outputs saved to /output/ folder") print("📊 Dashboard ready for Power BI / Tableau")
